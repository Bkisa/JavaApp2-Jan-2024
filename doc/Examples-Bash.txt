#---------------------------------------------------------------------------------------------------------

#!/bin/bash

echo $1



#---------------------------------------------------------------------------------------------------------
#   read fonksiyonu ile stdin'den değer okunabilir. bash içerisinde + işleminin sayı toplama olarak ele
#   yorumlanması için aşağıdaki gibi kullanılması gerekir
#---------------------------------------------------------------------------------------------------------

#!/bin/bash

echo "İki sayı giriniz:"
read a
read b

c=$(($a+$b))

echo "$a + $b = $c"



#---------------------------------------------------------------------------------------------------------

#!/bin/bash

read a
read b

c=$(($a*$b))

echo "$a * $b = $c"


#---------------------------------------------------------------------------------------------------------
#  Değişken let ile bildirilirsebu durumda işlemlerde parantez gerekmez
#---------------------------------------------------------------------------------------------------------

#!/bin/bash

echo "İki sayı giriniz:"
read a
read b


let sum=$a+$b

echo "$a + $b = $sum"

let mul=$a*$b

echo "$a * $b = $mul"


#---------------------------------------------------------------------------------------------------------
#  bash'de kuvvet alma işlemi ** operatörü ile yapılabilir
#---------------------------------------------------------------------------------------------------------

#!/bin/bash

echo "İki sayı giriniz:"
read a
read b


let pow=$a**$b

echo "$a ** $b = $pow"


#---------------------------------------------------------------------------------------------------------
#  bash'de fonksiyon yazmak için function anahtar sözcüğü kullanılır
#---------------------------------------------------------------------------------------------------------

#!/bin/bash

function main {
    echo "İki sayı giriniz:"
    read a
    read b

    let pow=$a**$b

    echo "$a ** $b = $pow"
}

main

#!/bin/bash

function main {
    echo "İki sayı giriniz:"
    read a
    read b

    printPower $a $b
}

function printPower {
    v1=$1
    v2=$2

    let pow=$v1**$v2

    echo "$a ** $b = $pow"
}

main
